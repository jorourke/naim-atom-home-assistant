```python
"""State management for Naim Media Player."""

import asyncio
import logging
import time
from enum import Enum, IntEnum
from typing import Callable

from homeassistant.components.media_player import MediaPlayerState

_LOGGER = logging.getLogger(__name__)

# Fields that are subject to debouncing (user action → ignore device echo)
DEBOUNCED_FIELDS = {"volume", "muted"}


class MediaInfo:
    """Handle media information."""

    def __init__(self):
        """Initialize media information."""
        self.title = None
        self.artist = None
        self.album = None
        self.duration = None
        self.image_url = None
        self.position = None


class NaimTransportState(IntEnum):
    """Enum for transport states (integer values from device)."""

    STOPPED = 1
    PLAYING = 2
    PAUSED = 3


class TransportStateString(str, Enum):
    """Enum for transport states (string values from WebSocket)."""

    PLAYING = "playing"
    PAUSED = "paused"
    STOPPED = "stopped"


TRANSPORT_STATES_STRING_LOOKUP = {
    TransportStateString.PLAYING: MediaPlayerState.PLAYING,
    TransportStateString.PAUSED: MediaPlayerState.PAUSED,
    TransportStateString.STOPPED: MediaPlayerState.IDLE,
}

NAIM_TRANSPORT_STATE_TO_HA_STATE = {
    NaimTransportState.PLAYING: MediaPlayerState.PLAYING,
    NaimTransportState.PAUSED: MediaPlayerState.PAUSED,
    NaimTransportState.STOPPED: MediaPlayerState.IDLE,
}

# Map of media_info attribute names for nested updates
MEDIA_INFO_FIELDS = {"title", "artist", "album", "duration", "image_url", "position"}


class NaimPlayerState:
    """Single source of truth for Naim device state.

    Both WebSocket and HTTP polling write here through update().
    Debouncing prevents device echoes from overwriting recent user actions.
    """

    def __init__(
        self,
        on_change: Callable[[], None] | None = None,
        debounce_timeout: float = 2.0,
    ):
        """Initialize state."""
        self._lock = asyncio.Lock()
        self._on_change = on_change
        self._debounce_timeout = debounce_timeout
        self._debounce_timestamps: dict[str, float] = {}

        # Device state
        self.power_state = MediaPlayerState.OFF
        self.playing_state = MediaPlayerState.IDLE
        self.volume = 0.0
        self.muted = False
        self.source = None
        self.available = True
        self.media_info = MediaInfo()

    async def update(self, source: str = "poll", **kwargs) -> None:
        """Update state fields.

        Args:
            source: Origin of update — "user", "websocket", or "poll".
                    User updates set debounce timestamps.
                    WebSocket/poll updates skip debounced fields.
        """
        current_time = time.time()
        changed = False

        async with self._lock:
            for key, value in kwargs.items():
                # Remap source_name to source to avoid conflict with the source parameter
                attr_name = "source" if key == "source_name" else key

                # Determine if this is a media_info field or a direct field
                if attr_name in MEDIA_INFO_FIELDS:
                    target = self.media_info
                elif hasattr(self, attr_name) and attr_name != "media_info":
                    target = self
                else:
                    continue

                # Check debounce for non-user sources
                if source != "user" and attr_name in DEBOUNCED_FIELDS:
                    last_user_action = self._debounce_timestamps.get(attr_name, 0)
                    if current_time - last_user_action < self._debounce_timeout:
                        _LOGGER.debug(
                            "Skipping %s update for '%s' (debounce active, %s source)",
                            source,
                            attr_name,
                            source,
                        )
                        continue

                # Set debounce timestamp for user actions
                if source == "user" and attr_name in DEBOUNCED_FIELDS:
                    self._debounce_timestamps[attr_name] = current_time

                # Apply update if value changed
                old_value = getattr(target, attr_name if target is self else key)
                if old_value != value:
                    if target is self:
                        setattr(self, attr_name, value)
                    else:
                        setattr(self.media_info, key, value)
                    changed = True

        if changed and self._on_change:
            self._on_change()

    @property
    def state(self) -> MediaPlayerState:
        """Get player state."""
        if self.power_state == MediaPlayerState.OFF:
            return MediaPlayerState.OFF
        if self.playing_state == MediaPlayerState.IDLE:
            return self.power_state
        return self.playing_state

    @property
    def media_title(self) -> str | None:
        return self.media_info.title

    @property
    def media_artist(self) -> str | None:
        return self.media_info.artist

    @property
    def media_album(self) -> str | None:
        return self.media_info.album

    @property
    def media_duration(self) -> int | None:
        return self.media_info.duration

    @property
    def media_position(self) -> int | None:
        return self.media_info.position

    @property
    def media_image_url(self) -> str | None:
        return self.media_info.image_url
```

```python
"""Device communication client for Naim Media Player."""

import asyncio
import json
import logging
import random
from typing import Any

import aiohttp
from homeassistant.components.media_player import MediaPlayerState
from homeassistant.core import HomeAssistant
from homeassistant.helpers.aiohttp_client import async_get_clientsession

from .exceptions import NaimConnectionError
from .state import (
    NAIM_TRANSPORT_STATE_TO_HA_STATE,
    TRANSPORT_STATES_STRING_LOOKUP,
    NaimPlayerState,
)

_LOGGER = logging.getLogger(__name__)

MAX_BUFFER_SIZE = 65536  # 64KB


class NaimClient:
    """Consolidated HTTP API and WebSocket client for Naim devices."""

    def __init__(
        self,
        hass: HomeAssistant,
        host: str,
        http_port: int,
        ws_port: int,
        state: NaimPlayerState,
        request_timeout: float = 10.0,
        max_retries: int = 3,
        ws_reconnect_interval: int = 5,
    ):
        self._hass = hass
        self._host = host
        self._http_port = http_port
        self._ws_port = ws_port
        self._state = state
        self._session = async_get_clientsession(hass)
        self._request_timeout = request_timeout
        self._max_retries = max_retries
        self._ws_reconnect_interval = ws_reconnect_interval
        self._ws_task = None
        self._connected = False
        self._buffer = ""
        self._decoder = json.JSONDecoder()

    # ========================================================================
    # HTTP API Methods
    # ========================================================================

    async def get_value(self, endpoint: str, variable: str) -> Any:
        """Get value from API endpoint with retries."""
        for retry in range(self._max_retries):
            try:
                url = f"http://{self._host}:{self._http_port}/{endpoint}"
                async with self._session.get(url, timeout=self._request_timeout) as response:
                    if response.status == 200:
                        data = await response.json()
                        return data.get(variable)
                    else:
                        _LOGGER.error("Error response %s from GET %s", response.status, url)
                if retry < self._max_retries - 1:
                    await asyncio.sleep(2**retry)
            except asyncio.TimeoutError:
                _LOGGER.warning(
                    "Timeout getting %s from %s (attempt %d/%d)",
                    variable, endpoint, retry + 1, self._max_retries,
                )
                if retry < self._max_retries - 1:
                    await asyncio.sleep(2**retry)
            except aiohttp.ClientError as error:
                raise NaimConnectionError(
                    f"Failed to get {variable} from {endpoint}: {error}"
                ) from error
        raise NaimConnectionError(
            f"Failed to get {variable} from {endpoint} after {self._max_retries} attempts"
        )

    async def set_value(self, endpoint: str, params: dict[str, str | int | bool]) -> bool:
        """Set value through API endpoint with retries."""
        for retry in range(self._max_retries):
            try:
                url = f"http://{self._host}:{self._http_port}/{endpoint}"
                async with self._session.put(url, params=params, timeout=self._request_timeout) as response:
                    if response.status == 200:
                        return True
                    else:
                        _LOGGER.error("Error response %s from PUT %s", response.status, url)
                if retry < self._max_retries - 1:
                    await asyncio.sleep(2**retry)
            except asyncio.TimeoutError:
                _LOGGER.warning("Timeout setting values to %s (attempt %d/%d)", endpoint, retry + 1, self._max_retries)
                if retry < self._max_retries - 1:
                    await asyncio.sleep(2**retry)
            except aiohttp.ClientError as error:
                raise NaimConnectionError(f"Failed to set values to {endpoint}: {error}") from error
        raise NaimConnectionError(f"Failed to set values to {endpoint} after {self._max_retries} attempts")

    async def send_command(self, endpoint: str, cmd: str, **params) -> bool:
        """Send command to device with retries."""
        command_params = {"cmd": cmd, **params}
        for retry in range(self._max_retries):
            try:
                url = f"http://{self._host}:{self._http_port}/{endpoint}"
                async with self._session.get(url, params=command_params, timeout=self._request_timeout) as response:
                    if response.status == 200:
                        return True
                    else:
                        _LOGGER.error("Error response %s from command %s to %s", response.status, cmd, url)
                if retry < self._max_retries - 1:
                    await asyncio.sleep(2**retry)
            except asyncio.TimeoutError:
                _LOGGER.warning("Timeout sending command %s to %s (attempt %d/%d)", cmd, endpoint, retry + 1, self._max_retries)
                if retry < self._max_retries - 1:
                    await asyncio.sleep(2**retry)
            except aiohttp.ClientError as error:
                raise NaimConnectionError(f"Failed to send command {cmd} to {endpoint}: {error}") from error
        raise NaimConnectionError(f"Failed to send command {cmd} to {endpoint} after {self._max_retries} attempts")

    async def select_input(self, input_id: str) -> bool:
        """Select input source."""
        return await self.send_command(f"inputs/{input_id}", "select")

    # ========================================================================
    # High-Level Commands (write to state with source="user")
    # ========================================================================

    async def set_volume(self, volume: int) -> None:
        """Set device volume (0-100)."""
        await self._state.update(source="user", volume=volume / 100)
        await self.set_value("levels/room", {"volume": volume})

    async def set_mute(self, mute: bool) -> None:
        """Set device mute state."""
        await self._state.update(source="user", muted=mute)
        await self.set_value("levels/room", {"mute": int(mute)})

    async def set_power(self, on: bool) -> None:
        """Set device power state."""
        power_value = "on" if on else "lona"
        await self.set_value("power", {"system": power_value})

    async def send_playback_command(self, cmd: str) -> None:
        """Send playback command (playpause, next, prev, seek)."""
        await self.send_command("nowplaying", cmd)

    # ========================================================================
    # Polling
    # ========================================================================

    async def poll_state(self) -> None:
        """Fetch full device state via HTTP. Writes to state with source='poll'."""
        try:
            power = await self._get_value_safe("power", "system")
            if power is None:
                if self._state.available:
                    _LOGGER.debug("Device %s is now unavailable", self._host)
                await self._state.update(source="poll", available=False)
                return

            if not self._state.available:
                _LOGGER.info("Device %s is now available", self._host)
            await self._state.update(source="poll", available=True)

            if power == "lona":
                await self._state.update(source="poll", power_state=MediaPlayerState.OFF)
                return

            # Device is on — fetch full state
            transport = await self._get_value_safe("nowplaying", "transportState")
            ha_state = NAIM_TRANSPORT_STATE_TO_HA_STATE.get(transport, MediaPlayerState.ON)
            await self._state.update(source="poll", power_state=MediaPlayerState.ON, playing_state=ha_state)

            # Volume and mute
            volume = await self._get_value_safe("levels/room", "volume")
            if volume is not None:
                await self._state.update(source="poll", volume=int(volume) / 100)
            mute = await self._get_value_safe("levels/room", "mute")
            if mute is not None:
                await self._state.update(source="poll", muted=bool(int(mute)))

            # Media info
            title = await self._get_value_safe("nowplaying", "title")
            artist = await self._get_value_safe("nowplaying", "artistName")
            album = await self._get_value_safe("nowplaying", "albumName")
            duration = await self._get_value_safe("nowplaying", "duration")
            position = await self._get_value_safe("nowplaying", "transportPosition")
            artwork = await self._get_value_safe("nowplaying", "artwork")
            device_source = await self._get_value_safe("nowplaying", "source")

            await self._state.update(
                source="poll",
                title=title,
                artist=artist,
                album=album,
                duration=duration,
                position=position,
                image_url=artwork,
            )

            if device_source:
                await self._state.update(source="poll", source_name=device_source)

        except aiohttp.ClientError as error:
            _LOGGER.debug("Error polling state for %s: %s", self._host, error)
            await self._state.update(source="poll", available=False)

    async def _get_value_safe(self, endpoint: str, variable: str) -> Any:
        """Get value without raising — returns None on failure."""
        try:
            timeout = aiohttp.ClientTimeout(total=5)
            url = f"http://{self._host}:{self._http_port}/{endpoint}"
            r = await self._session.get(url, timeout=timeout)
            if r.status == 200:
                data = await r.json()
                return data.get(variable)
        except (aiohttp.ClientError, asyncio.TimeoutError) as error:
            _LOGGER.debug("Cannot reach device at %s: %s", self._host, error)
        return None

    # ========================================================================
    # WebSocket
    # ========================================================================

    async def start_websocket(self) -> None:
        """Start the WebSocket listener task."""
        self._ws_task = asyncio.create_task(self._socket_listener())

    async def stop_websocket(self) -> None:
        """Stop the WebSocket listener task."""
        if self._ws_task and not self._ws_task.done():
            self._ws_task.cancel()
            try:
                done, pending = await asyncio.wait([self._ws_task], timeout=1.0)
                if pending:
                    _LOGGER.warning("Timeout waiting for WebSocket to close")
            except Exception as err:
                _LOGGER.error("Error stopping WebSocket: %s", err)

    async def _socket_listener(self) -> None:
        """Listen to TCP socket updates from the device."""
        _LOGGER.info("Starting WebSocket listener at %s:%d", self._host, self._ws_port)

        while True:
            writer = None
            self._buffer = ""  # Clear buffer on each connection attempt (Issue #5)
            try:
                reader, writer = await asyncio.open_connection(self._host, self._ws_port)
                self._connected = True
                _LOGGER.info("WebSocket connected to %s:%d", self._host, self._ws_port)

                # Connection restored — do a full state refresh
                await self.poll_state()

                while True:
                    try:
                        data = await reader.read(4096)
                        if not data:
                            _LOGGER.warning("WebSocket connection closed by server")
                            break

                        self._buffer += data.decode("utf-8")

                        # Buffer overflow protection
                        if len(self._buffer) > MAX_BUFFER_SIZE:
                            _LOGGER.warning("WebSocket buffer exceeded %d bytes, clearing", MAX_BUFFER_SIZE)
                            self._buffer = ""
                            continue

                        while self._buffer:
                            try:
                                obj, idx = self._decoder.raw_decode(self._buffer)
                                message_str = self._buffer[:idx]
                                self._buffer = self._buffer[idx:]
                                await self._handle_message(message_str)
                            except json.JSONDecodeError:
                                break
                    except Exception as error:
                        _LOGGER.error("Error receiving WebSocket data: %s", error)
                        break

            except Exception as error:
                self._connected = False
                _LOGGER.debug(
                    "WebSocket connection failed (%s:%d): %s. Retrying...",
                    self._host, self._ws_port, error,
                )

            finally:
                self._connected = False
                if writer is not None:
                    try:
                        writer.close()
                        await writer.wait_closed()
                    except Exception as error:
                        _LOGGER.warning("Error closing WebSocket: %s", error)

            # Reconnect with jitter
            jitter = random.uniform(0, self._ws_reconnect_interval * 0.3)
            await asyncio.sleep(self._ws_reconnect_interval + jitter)

    async def _handle_message(self, raw_message: str) -> None:
        """Parse a WebSocket JSON message and update state."""
        try:
            all_state = json.loads(raw_message)
            state_data = all_state.get("data", {})

            updates = {}

            # Media info from trackRoles
            track_roles = state_data.get("trackRoles", {})
            media_data = track_roles.get("mediaData", {}).get("metaData", {})
            updates["title"] = track_roles.get("title")
            updates["artist"] = media_data.get("artist")
            updates["album"] = media_data.get("album")
            updates["image_url"] = track_roles.get("icon")

            # Playing state
            playing_state_str = state_data.get("state")
            if playing_state_str:
                updates["playing_state"] = TRANSPORT_STATES_STRING_LOOKUP.get(
                    playing_state_str, MediaPlayerState.IDLE
                )

            # Duration and position
            duration_ms = state_data.get("status", {}).get("duration")
            if duration_ms is not None:
                try:
                    updates["duration"] = float(duration_ms) / 1000
                except (ValueError, TypeError):
                    _LOGGER.warning("Invalid duration value: %s", duration_ms)

            position_ms = all_state.get("playTime", {}).get("i64_")
            if position_ms is not None:
                updates["position"] = position_ms / 1000

            # Source detection
            source = self._extract_source(state_data)
            if source:
                updates["source_name"] = source

            await self._state.update(source="websocket", **updates)

        except json.JSONDecodeError as error:
            _LOGGER.error("Error parsing WebSocket message: %s", error)
        except Exception as error:
            _LOGGER.error("Error handling WebSocket message: %s", error)

    def _extract_source(self, live_status: dict) -> str | None:
        """Extract source name from WebSocket status data."""
        media_roles = live_status.get("mediaRoles", {})
        if media_roles:
            meta = media_roles.get("mediaData", {}).get("metaData", {})
            if meta.get("serviceID") == "roon":
                return "Roon"

        source = media_roles.get("title") if media_roles else None
        if not source:
            context = live_status.get("contextPath")
            if context and context.startswith("spotify"):
                source = "Spotify"
        return source
```

```python
import asyncio
import logging
import random
import string

from homeassistant.components.media_player import (

    MediaPlayerState,

PLATFORMS = [Platform.MEDIA_PLAYER]

def round_to_nearest(val: float, step: float = 0.01) -> float:
    return round(val / step) * step


async def async_setup_entry(
    hass: HomeAssistant,
    entry: ConfigEntry,
    async_add_entities,
) -> None:
    """Set up the Naim Media Player from a config entry."""
    ip_address = entry.data[CONF_IP_ADDRESS]
    name = entry.data.get(CONF_NAME, DEFAULT_NAME)
    entity_id = entry.data.get("entity_id")
    volume_step = entry.data.get(CONF_VOLUME_STEP, DEFAULT_VOLUME_STEP)

    )

    DEFAULT_SOURCE_MAP = {
        "Analog 1": "ana1",
        "Digital 1": "dig1",
        "Digital 2": "dig2",
        "Digital 3": "dig3",
        "Bluetooth": "bluetooth",
        "Web Radio": "radio",
        "Spotify": "spotify",
        "Roon": "roon",
        "HDMI": "hdmi",
    }

        name: str,
        ip_address: str,
        entity_id: str | None = None,
        volume_step: float = DEFAULT_VOLUME_STEP,
        sources: dict[str, str] | None = None,
    ):
        _LOGGER.info("Initializing Naim media player: %s at %s", name, ip_address)
        self._hass = hass
        self._name = name
        self._ip_address = ip_address

        self._source_list = list(self._source_map.keys())

        )
        self._client = NaimClient(
            hass=hass,
            host=ip_address,
            http_port=15081,
            ws_port=4545,
            state=self._state,
        )
        asyncio.create_task(self._client.start_websocket())

    @property
    def available(self) -> bool:
        return self._state.available

    @property
    def supported_features(self) -> int:
        return (

            | MediaPlayerEntityFeature.TURN_OFF

            | MediaPlayerEntityFeature.NEXT_TRACK
            | MediaPlayerEntityFeature.PREVIOUS_TRACK
        )

    def state(self) -> MediaPlayerState | None:

    # Commands — delegate to client

    async def async_turn_on(self):

    async def async_set_volume_level(self, volume: float):
        volume = max(0.0, min(1.0, volume))
        volume = round_to_nearest(volume)
        await self._client.set_volume(int(volume * 100))

        await self._client.send_playback_command("prev")

    async def async_select_source(self, source):
        if source in self._source_map:
            await self._client.select_input(self._source_map[source])

    async def async_will_remove_from_hass(self):
        await self._client.stop_websocket()
```
